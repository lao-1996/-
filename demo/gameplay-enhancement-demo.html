// 初始化游戏性增强系统
const difficultySystem = new DynamicDifficultySystem();
const aiManager = new AIManager();
const itemSystem = new ItemSystem();

// 游戏主循环更新
function gameUpdate(deltaTime) {
    // 更新动态难度系统
    difficultySystem.update(deltaTime, gameState);
    const difficultyParams = difficultySystem.getCurrentDifficultyParams();
    
    // 更新AI系统（使用当前难度参数）
    aiManager.setDifficultyFactor(difficultyParams.enemyAILevel);
    aiManager.update(deltaTime, gameState);
    
    // 更新道具系统
    itemSystem.setDifficultyFactor(difficultyParams.itemSpawnRate);
    itemSystem.update(deltaTime, gameState);
    
    // 获取道具使用建议
    const recommendations = itemSystem.getItemRecommendations(gameState);
    if (recommendations.topRecommendation) {
        showItemSuggestion(recommendations.topRecommendation);
    }
}

// 处理玩家动作
function onPlayerAction(action, data) {
    // 记录到难度系统
    difficultySystem.recordAction(action, data);
    
    // 记录到AI分析系统
    if (aiManager.globalPlayerAnalyzer) {
        aiManager.globalPlayerAnalyzer.recordAction(
            action,
            gameState.player.position,
            gameState.time,
            data
        );
    }

    }
}
